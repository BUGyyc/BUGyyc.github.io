冒泡排序

public int[] Sort(int[] array){
    int[] arr = Arrays.copyOf(array,array.length);
    for(int i = 1;i<arr.length;i++){
        boolean res = true;//标记，判断是否有交换
        for(int j = 1;j<arr.length-i;j++){
            if(arr[j] > arr[j+1]){
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
                res = false;
            }
        }

        if(res){
            break;
        }
    }
    return arr;
}

选择排序

先找到堆中，找到最大（或最小）的一项，拎出来，然后再从剩下的堆中找到最大（或最小）的一项，直到堆中没有数据，这样所有的数据就有序了。

public int[] Sort(int[] array){
    int[] arr = Arrays.CopyOf(array,array.length);
    for(int i = 0;i<arr.length - 1;i++){
        int min = i;
        for(int j = i+1;j<arr.length;j++){
            if(arr[j] < arr[min]){
                min = j;
            }
        }

        if(min != i){
            int temp = arr[i];
            arr[i] = arr[min];
            arr[min] = temp;
        }
    }
    return arr;
}
