队列实现堆栈

public class Stack{

    private Queue<Integer> q1 = new Queue<Integer>();
    private Queue<Integer> q2 = new Queue<Integer>();

    private int top;

    public void Push(int x){
        q1.add(x);
        top = x;
    }

    public int Pop(){
        while(q1.size()>1){
            top = q1.remove();
            q2.add(top);
        }

        q1.remove();
        Queue<Integer> temp = q1;
        q1 = q2;
        q2 = temp;
    }

}
